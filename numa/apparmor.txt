#include <tunables/global>

profile numa flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities required by Numa
  # NET_BIND_SERVICE: bind to privileged ports (DNS :53, DoT :853)
  # KILL: send signals to child processes (graceful reload/restart)
  # SYS_RESOURCE: manage resource limits
  capability net_bind_service,
  capability kill,
  capability sys_resource,
  capability dac_override,
  capability dac_read_search,
  # SETUID/SETGID: nginx 'user' directive calls setuid/setgid even when
  # already running as root (needed if user directive is present in config)
  capability setuid,
  capability setgid,

  # --- Container startup ---
  /init rix,
  /bin/sh rix,

  # --- Networking ---
  network inet  stream,
  network inet  dgram,
  network inet6 stream,
  network inet6 dgram,
  network netlink raw,

  # --- Numa binary ---
  /usr/local/bin/numa            mrix,

  # --- Configuration (addon_config mapping → /config inside container) ---
  /config/                       r,
  /config/numa.toml              rw,

  # --- Persistent data volume (/data inside container) ---
  # Stores TLS CA material, self-signed certs, internal state
  /data/                         r,
  /data/numa/                    rw,
  /data/numa/**                  rw,

  # --- Bundled default config (read-only, embedded in image) ---
  /etc/numa/                     r,
  /etc/numa/**                   r,

  # --- nginx binary (Alpine installs to /usr/sbin/nginx) ---
  /usr/sbin/nginx                mrix,

  # --- nginx config, templates (read) and runtime-generated files (write) ---
  # tempio writes upstream.conf and ingress.conf into /etc/nginx at startup.
  /etc/nginx/                    r,
  /etc/nginx/**                  rw,

  # --- nginx runtime state and default log directory ---
  # nginx opens /var/lib/nginx/logs/error.log as its built-in default error
  # log BEFORE reading nginx.conf (so error_log /proc/1/fd/1 in the config
  # doesn't help for this first open). Both paths must be writable.
  /var/lib/nginx/                rw,
  /var/lib/nginx/**              rw,
  /var/log/nginx/                rw,
  /var/log/nginx/**              rw,

  # --- User/group resolution (getpwnam/getgrnam called by nginx) ---
  # abstractions/base does NOT include passwd/group — add them explicitly.
  /etc/passwd                    r,
  /etc/group                     r,
  /etc/nsswitch.conf             r,

  # --- Write to parent-process file descriptors ---
  # nginx error_log is redirected to /proc/1/fd/1 (container stdout).
  # Only PID 1's stdout fd needs to be writable — granting rw to all process
  # fds would be unnecessarily broad. Directory-read on any pid's fd dir is
  # kept for tools that introspect their own open fds (e.g. s6, bashio).
  # NOTE: child-profile scoping of setuid/setgid (CodeRabbit nitpick) is
  # intentionally not implemented — HA Supervisor loads AppArmor profiles as
  # flat files; Cx child profiles would require duplicating all nginx rules and
  # are not used by any official HA add-on. Risk is minimal: Numa is a
  # single-purpose DNS binary with no reason to call setuid/setgid.
  @{PROC}/[0-9]*/fd/             r,
  @{PROC}/1/fd/1                 rw,

  # S6-Overlay
  /init rix,
  /bin/** rix,
  /usr/bin/** rix,
  /run/{s6,s6-rc*,service}/** rix,
  /package/** rix,
  /command/** rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/s6-overlay/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # --- System libraries and dynamic linker ---
  /lib/**                        mr,
  /lib64/**                      mr,
  /usr/lib/**                    mr,
  /usr/local/lib/**              mr,

  # --- /proc access needed by Tokio async runtime ---
  @{PROC}/                       r,
  @{PROC}/@{pid}/                r,
  @{PROC}/@{pid}/fd/             r,
  @{PROC}/@{pid}/maps            r,
  @{PROC}/@{pid}/status          r,
  @{PROC}/@{pid}/task/           r,
  @{PROC}/sys/kernel/hostname    r,
  @{PROC}/sys/net/core/somaxconn r,

  # --- DNS resolver bootstrap (upstream DoH/DoT TLS handshake) ---
  /etc/resolv.conf               r,
  /etc/hosts                     r,
  /etc/ssl/certs/**              r,
  /etc/ca-certificates/**        r,
  /usr/share/ca-certificates/**  r,

  # --- Timezone data (timestamps in query log) ---
  /usr/share/zoneinfo/**         r,
  /etc/localtime                 r,

  # --- Temporary files (tarball extraction during image build is done
  #     at build time; runtime tmp is used by rcgen for cert generation) ---
  /tmp/                          r,
  /tmp/**                        rw,

  # --- Deny everything else ---
  deny /proc/sysrq-trigger       w,
  deny /proc/sys/fs/binfmt_misc  w,
  deny @{PROC}/sys/kernel/sysrq  w,
  deny /sys/firmware/**          rw,
}
